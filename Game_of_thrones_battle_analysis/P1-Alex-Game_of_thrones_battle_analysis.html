#  split_col 
def split_col(origin_df,col_name,split_key):
    s = origin_df[col_name].str.split(split_key, expand=True).stack()
    i = s.index.get_level_values(0)
    df2 = origin_df.loc[i].copy()
    df2[col_name] = s.values
    return df2
